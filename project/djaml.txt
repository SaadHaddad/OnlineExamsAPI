import axios from "axios";

// mothods has to retrun a {error, payload} object like

const base_url = process.env.BASE_URL;

const getData = async (url, config) => {
    let error = false;
    let payload = false;

    const response = await axios.get(${ base_url }${ url }, config);

    if (response.status === 200 || response.status === 201) {
        payload = response.data;
    } else {
        error = response.status;
    }

    return { payload, error };
};

const postData = async (url, body, config) => {
    let error = false;
    let payload = false;

    const response = await axios.post(${ base_url }${ url }, body, config);

    if (response.status === 200 || response.status === 201) {
        payload = response.data;
    } else {
        error = response.status;
    }

    return { payload, error };
};

const putData = async (url, body, config) => {
    const response = await axios.put(${ base_url }${ url }, body, config);

    return response;
};

const deleteData = async (url, config) => {
    const response = await axios.delete(url, config);

    return response;
};

const patchData = async (url, body, config) => {
    const response = await axios.patch(${ base_url }${ url }, body, config);

    return response;
};

export { getData, postData, putData, deleteData, patchData };
